void sort1(int& a,int& b,int& c){
  int s[3]={a,b,c};
  int temp;
  for(int i=0;i<2;i++)
  for(int j=0;j<2-i;j++){
    if(s[j]<s[j+1]){
        temp=s[j];
        s[j]=s[j+1];
        s[j+1]=temp;
    }
  }

}
