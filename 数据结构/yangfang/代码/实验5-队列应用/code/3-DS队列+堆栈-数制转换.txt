#include<iostream>
#include<stack>
#include<queue>
using namespace std;

int main() {
    double n,decimal;
    int k,intenger,i,temp;
    int t;
    char ch;

    cin>>t;
    while(t--){
        stack <char> s;
        queue<char> que;

        cin>>n>>k;
        intenger=(int)n;   //获得整数部分
        decimal=n-intenger; //获得小数部分

        while(intenger){   //处理整数部分
            temp=intenger%k;
            if(temp>=10&&k>10)  //注意不同进制的处理
                ch=temp-10+'A';
            else
                ch=temp+'0';
            s.push(ch);
            intenger=intenger/k;
        }

        i=0;
        while(i<3 && decimal!=0){  //处理小数部分，i控制小数位数为3位
            temp=(int)(decimal*k);
            if(temp>=10&&k>10)
                ch=temp-10+'A';
            else
                ch=temp+'0';
            que.push(ch);
            decimal=decimal*k-temp;
            i++;
        }
        for(int j=3-i;j>0;j--)  //若小数点后不足3位，补0
            que.push('0');

        while(!s.empty()){   //输出整数部分
            cout<<s.top();
            s.pop();
        }
        cout<<".";
        while(!que.empty()){  //输出小数部分
            cout<<que.front();
            que.pop();
        }
        cout<<endl;
    }
    return 0;
}
