#include<iostream>
#include<queue>
using namespace std;
int main() {
    queue<int> q[2];
    int t,data;
    cin>>t;

    while(t--){   //数据入队列，偶数进q[0]，奇数进q[1]
      cin>>data;
      q[data%2].push(data);
    }

    int i=1;
    if(q[1].empty())    //输出的第一个数据格式和其他不同，单独处理
        i=0;
    cout<<q[i].front();
    q[i].pop();

    i=1;
    while(!q[0].empty()|| !q[1].empty()){
      i++;
      if(!q[1].empty()){  //每隔一次输出一个数据
          cout<<" "<<q[1].front();
          q[1].pop();
      }
      if(!q[0].empty() && i%2==0){ //每隔两次输出一个数据
          cout<<" "<<q[0].front();
          q[0].pop();
      }
    }

    return 0;
}
