#include<iostream>
using namespace std;

struct ListNode{//结点类
    int data;
    ListNode *next;
};

class Linklist{
    ListNode *head;
public:
    Linklist(){
        int len,e;
        cin>>len;
        head=new ListNode{len,NULL};
        ListNode *p=head;
        for(int i=0;i<len;i++){
           cin>>e;
           ListNode *s=new ListNode{e,NULL};
           p->next=s;
           p=s;
        }
    }

    ~Linklist(){
        ListNode *p;
        while(head){
           p=head;
           head=p->next;
           delete p;
        }
    }

    void L_del(){ //双循环：为每个结点遍历后面的结点，看是否有重复的元素
        ListNode *p,*q,*s;
        p=head->next;
        while(p){   //p指向当前结点
            q=p;
            while(q->next){  //用指针q遍历p后面的每个结点，为了删除操作，q指向待比较元素的前一个结点
                if(q->next->data==p->data){
                    s=q->next;  //s指向待删除结点
                    q->next=s->next;
                    head->data--;
                    delete s;
                }
                else
                    q=q->next;
            }
            p=p->next;
        }
    }

    void L_display(){
        ListNode *p=head->next;
        cout<<head->data<<":";
        while(p){
          cout<<" "<<p->data;
          p = p->next;
        }
        cout<<endl;
    }
};

int main(){
    int t;
    cin>>t;
    while(t--){
        Linklist l;
        l.L_del();
        l.L_display();
    }
    return 0;
}
