第一种：使用一级指针
#include<iostream>
using namespace std;
int main()
{
    int p[2][3],q[3][2],*k;
    int t;
    cin>>t;

    while(t--){
      for(int i=0;i<2;i++)
        for(int j=0;j<3;j++)
          cin>>p[i][j];

      k=q[0];
      for(int i=0;i<3;i++){
        for(int j=0;j<2;j++)
           *k++=*(*(p+j)+2-i);  //等价写法*k++=p[j][2-i]
      }

      k=q[0];
      for(int i=0;i<3;i++){
        for(int j=0;j<2;j++)
          cout<<*k++<<" ";
        cout<<endl;
      }
   }
   return 0;
}

第二种：直接使用数组名处理，相当于行指针的用法
#include<iostream>
using namespace std;
int main()
{
    int p[2][3],q[3][2];
    int t;
    cin>>t;

    while(t--){
      for(int i=0;i<2;i++)
        for(int j=0;j<3;j++)
          cin>>p[i][j];

      for(int i=0;i<3;i++){
        for(int j=0;j<2;j++)
           *(*(q+i)+j)=*(*(p+j)+2-i);     // 等价写法：q[i][j]=p[j][2-i];
      }

      for(int i=0;i<3;i++){
        for(int j=0;j<2;j++)
          cout<<*(*(q+i)+j)<<" ";     //等价写法：cout<<q[i][j]<<" ";
        cout<<endl;
      }
  }

   return 0;
}
