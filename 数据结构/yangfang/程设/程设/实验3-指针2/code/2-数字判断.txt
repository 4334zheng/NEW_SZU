int isNumber(char *s)
{
	char *p=s;
	if(s[0]=='-'|| s[0]=='+')  //若有正负符号，则p后移一位
		p++;

	while(*p)
	{
		if(!(isdigit(*p)))
		   return -1;
                               p++;
	}
	return atoi(s);
}
