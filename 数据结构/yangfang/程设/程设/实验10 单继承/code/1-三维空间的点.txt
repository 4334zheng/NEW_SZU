#include<iostream>
#include<cmath>
using namespace std;

class C2D
{
protected:
    double x, y;
public:
    C2D(double x1, double y1): x(x1), y(y1) {}
    double getDistance()
    {   return sqrt(x*x+y*y);}
};

class C3D: public C2D{
private:
    double z;
public:
    C3D(double x1, double y1, double z1): C2D(x1, y1), z(z1) {}
    double getDistance()
    {   return sqrt(x*x+y*y+z*z);}

};

int main()
{   
    double x, y, z;

    cin>>x>>y;
    C2D p1(x, y);
    cout<<p1.getDistance()<<endl;

    cin>>x>>y>>z;
    C3D p2(x, y, z);
    cout<<p2.getDistance()<<endl;

    cin>>x>>y>>z;
    C3D p3(x, y, z);
    cout<<p3.getDistance()<<endl;

    p1=p3;
    cout<<p1.getDistance()<<endl;

    return 0;
}
