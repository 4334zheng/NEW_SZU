#include<iostream>
using namespace std;

class Student {
   int id;
   int score;
   static int  maxscore;//最高分数
   static int maxid;//最高分数学生学号
public:
   Student(int ti=0,int ts=0):id(ti),score(ts){}
   static void findMax(Student& stu){ //寻找最高成绩和学号
      if(stu.score>maxscore){
         maxscore=stu.score;
         maxid=stu.id;
      }
   }
   static int getMaxScore(){//返回最高成绩
       return maxscore;
   }
   static int getMaxID(){//返回最高成绩的学号
       return maxid;
   }
};

int Student::maxscore=0;
int Student::maxid=0;

int main(){
   int t,id,score;
   cin>>t;
   while(t--){
      cin>>id>>score;
      Student stu(id,score);
      Student::findMax(stu);
   }
   cout<<Student::getMaxID()<<"--"<<Student::getMaxScore()<<endl;

}
